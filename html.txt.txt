        let currentLanguage = 'ar';

        function switchLanguage(lang) {
            currentLanguage = lang;
            document.querySelectorAll('.i18n').forEach(element => {
                const key = element.getAttribute('data-key');
                if (translations[lang][key]) {
                    element.textContent = translations[lang][key];
                }
            });
            
            // تحديث أزرار اللغة
            document.querySelectorAll('.lang-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            document.querySelector(`.lang-btn[onclick="switchLanguage('${lang}')"]`).classList.add('active');
            
            // تحديث اتجاه الصفحة
            document.documentElement.dir = lang === 'ar' ? 'rtl' : 'ltr';
            document.documentElement.lang = lang;
        }

        // نظام إدارة النماذج والواجهة
        const app = {
            currentUser: null,
            
            init() {
                this.setupEventListeners();
                this.loadUserData();
                this.setupScrollAnimations();
            },
            
            setupEventListeners() {
                // التنقل السلس
                const scrollLinks = document.querySelectorAll('a[href^="#"]');
                scrollLinks.forEach(link => {
                    link.addEventListener('click', this.smoothScroll);
                });
                
                // نماذج التسجيل
                document.getElementById('login-form').addEventListener('submit', this.handleLogin.bind(this));
                document.getElementById('register-form').addEventListener('submit', this.handleRegister.bind(this));
            },
            
            smoothScroll(e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetElement = document.querySelector(targetId);
                
                if (targetElement) {
                    window.scrollTo({
                        top: targetElement.offsetTop - 80,
                        behavior: 'smooth'
                    });
                }
            },
            
            setupScrollAnimations() {
                const observerOptions = {
                    threshold: 0.1,
                    rootMargin: '0px 0px -50px 0px'
                };
                
                const observer = new IntersectionObserver((entries) => {
                    entries.forEach(entry => {
                        if (entry.isIntersecting) {
                            entry.target.style.opacity = '1';
                            entry.target.style.transform = 'translateY(0)';
                        }
                    });
                }, observerOptions);
                
                // تطبيق التأثير على العناصر
                const animatedElements = document.querySelectorAll('.feature-card, .isteam-pillar, .pillar-card, .path-card');
                animatedElements.forEach(el => {
                    el.style.opacity = '0';
                    el.style.transform = 'translateY(30px)';
                    el.style.transition = 'opacity 0.6s ease, transform 0.6s ease';
                    observer.observe(el);
                });
            },
            
            handleLogin(e) {
                e.preventDefault();
                const email = document.getElementById('login-email').value;
                const password = document.getElementById('login-password').value;
                
                // محاكاة عملية تسجيل الدخول
                if (email && password) {
                    alert(currentLanguage === 'ar' ? 'تم تسجيل الدخول بنجاح!' : 'Login successful!');
                    closeModal('login-modal');
                    this.currentUser = { email, name: 'مستخدم' };
                    this.updateUI();
                }
            },
            
            handleRegister(e) {
                e.preventDefault();
                const formData = {
                    name: document.getElementById('register-name').value,
                    email: document.getElementById('register-email').value,
                    role: document.getElementById('register-role').value,
                    password: document.getElementById('register-password').value
                };
                
                // محاكاة عملية التسجيل
                if (this.validateRegister(formData)) {
                    alert(currentLanguage === 'ar' ? 'تم إنشاء الحساب بنجاح! مرحباً بك في منصة ابتكر عُمان' : 'Account created successfully! Welcome to Innovate Oman Platform');
                    closeModal('register-modal');
                    this.currentUser = formData;
                    this.updateUI();
                }
            },
            
            validateRegister(data) {
                // تحقق بسيط من البيانات
                if (!data.name || !data.email || !data.role || !data.password) {
                    alert(currentLanguage === 'ar' ? 'يرجى ملء جميع الحقول' : 'Please fill all fields');
                    return false;
                }
                return true;
            },
            
            updateUI() {
                if (this.currentUser) {
                    // تحديث الواجهة للمستخدم المسجل
                    const authButtons = document.querySelector('.auth-buttons');
                    authButtons.innerHTML = `
                        <div class="language-switcher">
                            <button class="lang-btn ${currentLanguage === 'ar' ? 'active' : ''}" onclick="switchLanguage('ar')">العربية</button>
                            <button class="lang-btn ${currentLanguage === 'en' ? 'active' : ''}" onclick="switchLanguage('en')">English</button>
                        </div>
                        <span style="color: white; margin: 0 1rem;">${currentLanguage === 'ar' ? 'مرحباً' : 'Welcome'} ${this.currentUser.name}</span>
                        <button class="btn btn-outline" onclick="app.logout()">${currentLanguage === 'ar' ? 'تسجيل الخروج' : 'Logout'}</button>
                    `;
                }
            },
            
            logout() {
                this.currentUser = null;
                location.reload();
            },
            
            loadUserData() {
                // تحميل بيانات المستخدم من localStorage
                const savedUser = localStorage.getItem('innovateOmanUser');
                if (savedUser) {
                    this.currentUser = JSON.parse(savedUser);
                    this.updateUI();
                }
            },
            
            saveUserData() {
                if (this.currentUser) {
                    localStorage.setItem('innovateOmanUser', JSON.stringify(this.currentUser));
                }
            }
        };

        // دوال إدارة النوافذ المنبثقة
        function showModal(modalId) {
            document.getElementById(modalId).classList.add('active');
        }

        function closeModal(modalId) {
            document.getElementById(modalId).classList.remove('active');
        }

        // إغلاق النافذة عند النقر خارجها
        document.querySelectorAll('.modal').forEach(modal => {
            modal.addEventListener('click', function(e) {
                if (e.target === this) {
                    this.classList.remove('active');
                }
            });
        });

        // تهيئة التطبيق عند تحميل الصفحة
        document.addEventListener('DOMContentLoaded', () => {
            app.init();
        });
